// 백준 13549 : 숨바꼭질 3 (BFS + 다익스트라)
#include <vector>
#include <string>
#include <queue>
#include <iostream>

using namespace std;

int main() {
	int N, K; // 수빈이가 있는 위치 N, 동생이 있는 위치 K
	cin >> N >> K;

	int min_time[100001]; // 각 좌표에 도달하는 최소시간을 저장 (사용인덱스 : 0~100000)
	for (int i = 0; i < 100001; i++) {
		min_time[i] = 100000;
	}

	queue<pair<int, int>> q;// {수빈의 위치, 해당 지점까지 걸린 시간}
	q.push({ N, 0 });
	min_time[N] = 0;

	while (!q.empty()) {
		int x = q.front().first;
		int time = q.front().second;
		q.pop();

		vector<int> nextX = { x + 1, x - 1, x * 2 };
		for (int i = 0; i < 3; i++) {
			if (0 <= nextX[i] && nextX[i] <= 100000) {

				int tmp_time = time;
				if (i != 2) tmp_time++;

				if (min_time[nextX[i]] > tmp_time) {
					min_time[nextX[i]] = tmp_time;
					q.push({ nextX[i], tmp_time });
				}
			}
		}
	}
	cout << min_time[K] << '\n';
}
