// 백준 7576 : 토마토
#include <vector>
#include <string>
#include<queue>
#include <iostream>

using namespace std;

queue<pair<int, int>> q;
bool chk[1001][1001];
int day[1001][1001];
int dx[4] = { 0, 1, 0, -1 };
int dy[4] = { 1, 0, -1, 0 };

int main() {
	int m, n; // 상자의 가로 칸의 수 m, 세로 칸의 수 n
	cin >> m >> n;

	vector<int> tomato[1001];
	int raw = 0;
	for (int i = 0; i < n; i++) {
		for (int j = 0; j < m; j++) {
			int tmp;
			cin >> tmp;
			tomato[i].push_back(tmp);
			if (tmp == 1) {
				q.push({ j, i });
				chk[i][j] = true;
				day[i][j] = 0;
			}
			else if (tmp == 0) {
				raw++;
			}
		}
	}

	// BFS 탐색
	while (!q.empty()) {
		int x = q.front().first;
		int y = q.front().second;
		q.pop();

		for (int i = 0; i < 4; i++) {
			int next_x = x + dx[i];
			int next_y = y + dy[i];
			
			if (next_y < 0 || next_y >= n || next_x < 0 || next_x >= m) continue;

			if (tomato[next_y][next_x] == 0 && chk[next_y][next_x] == 0) {
				q.push({ next_x , next_y });
				chk[next_y][next_x] = true;
				raw--;
				day[next_y][next_x] = day[y][x] + 1;
			}
		}
	}

	if (raw != 0) cout << -1 << endl;
	else {
		int maxDay = 0;
		for (int i = 0; i < n; i++) {
			for (int j = 0; j < m; j++) {
				if (maxDay < day[i][j]) maxDay = day[i][j];
			}
		}
		cout << maxDay << endl;
	}
}
